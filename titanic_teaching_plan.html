Q1
How do we load the Titanic CSV file and see the first 5 rows?
import pandas as pd df = pd.read_csv('titanic_dataset.csv') df.head() # first 5 rows df.head(10) # first 10 rows df.tail() # last 5 rows Tell students: head() is like peeking at the top of the register
Q2
How many passengers are in the dataset? How many columns?
df.shape # (rows, columns) # Output: (418, 12) # → 418 passengers, 12 columns → (418, 12) 418 passengers, 12 things we know about each one
Q3
What are the column names and their data types?
df.columns # list of column names df.dtypes # data type of each column df.info() # full summary: nulls + types info() is the fastest way to see the full picture of your data
Q4
Get basic statistics — average age, min/max fare, etc.
df.describe() # count, mean, std, min, max for numbers # Average fare: 35.6 | Max fare: 512 | Min age: 0.17 Ask students: why is min age 0.17? (babies were on the ship!)

df['Survived'].value_counts() # 0 → 266 did NOT survive # 1 → 152 survived 0: 266 1: 152 Survived=1 means alive, Survived=0 means did not survive

Session 2 — Selecting, Filtering & Sorting (Pandas)
[], loc, iloc, filtering with conditions, sort_values, value_counts
▼
Q6
Select only the Name, Age, and Sex columns.
df[['Name', 'Age', 'Sex']] # select multiple columns df['Age'] # select one column (Series) Single bracket [ ] = one column. Double bracket [[ ]] = multiple columns
Q7
Show only female passengers.
df[df['Sex'] == 'female'] # Returns only rows where Sex is female → 152 rows → 152 female passengers
Q8
Show passengers who survived AND were in 1st class.
df[(df['Survived'] == 1) & (df['Pclass'] == 1)] # & means AND | means OR ~ means NOT Always use ( ) around each condition when combining with & or |
Q9
Sort passengers by Fare from highest to lowest.
df.sort_values('Fare', ascending=False).head(5) # Top fare: 512 NRS — 1st class passengers paid the most
Q10
Select row 5 to row 10 using position.
df.iloc[5:10] # by position number (like list slicing) df.loc[5:10] # by index label df.iloc[0] # first row only iloc = position number (0,1,2...). loc = index label

Session 3 — Cleaning Missing Data (Pandas)
isnull, dropna, fillna, missing value strategy
▼
Q11
Find which columns have missing (NaN) values.
df.isnull().sum() # Age → 86 missing # Cabin → 327 missing (most passengers — not useful) # Fare → 1 missing Age:86 Cabin:327 Fare:1 Ask: why is Cabin missing for 327 people? (3rd class had shared rooms)
Q12
Fill missing Age values with the average age.
avg_age = df['Age'].mean() # calculate average print(avg_age) # ~30.27 df['Age'] = df['Age'].fillna(avg_age) # fill NaN with average df['Age'].isnull().sum() # → 0 (no more missing!) Strategy: fill numbers with mean or median. Never guess randomly.
Q13
Drop the Cabin column (too many missing values — not useful).
df = df.drop(columns=['Cabin']) # OR: drop rows with any missing value df_clean = df.dropna() # removes ALL rows with NaN Rule: if a column has >50% missing → consider dropping it
Q14
Add a new column — Age Group (Child, Adult, Senior).
def age_group(age): if age < 18: return 'Child' elif age < 60: return 'Adult' else: return 'Senior' df['AgeGroup'] = df['Age'].apply(age_group) df['AgeGroup'].value_counts() apply() runs a function on every row — very powerful!

Session 4 — Grouping & Aggregation (Pandas + NumPy)
groupby, agg, numpy arrays, mean, sum, std, unique
▼
Q15
What is the survival rate by passenger class?
df.groupby('Pclass')['Survived'].mean() # Pclass 1 → 0.63 (63% survived) # Pclass 2 → 0.47 (47% survived) # Pclass 3 → 0.24 (24% survived) Class 1: 63% Class 2: 47% Class 3: 24% Rich passengers (1st class) had much higher survival rate — shocking but true
Q16
What is the average fare paid by men vs women?
df.groupby('Sex')['Fare'].mean() # female → 44.5 male → 26.5
Q17
Show count, average age, and average fare by class.
df.groupby('Pclass').agg( total = ('PassengerId', 'count'), avg_age = ('Age', 'mean'), avg_fare = ('Fare', 'mean') )
Q18
NumPy basics — arrays and calculations.
import numpy as np ages = np.array(df['Age']) # convert to numpy array fares = np.array(df['Fare']) print(np.mean(ages)) # average age print(np.median(fares)) # middle fare value print(np.std(fares)) # how spread out fares are print(np.max(fares)) # highest fare: 512 NumPy is faster than Python lists for math on large data
Q19
How many unique ports did passengers board from?
df['Embarked'].unique() # S, C, Q df['Embarked'].nunique() # 3 df['Embarked'].value_counts() # S (Southampton) → 270 # C (Cherbourg) → 102 # Q (Queenstown) → 46

Session 5 — Visualisation with Matplotlib
bar chart, histogram, pie chart, scatter plot, line chart
▼
Q20
Bar chart — How many survived vs did not survive?
import matplotlib.pyplot as plt counts = df['Survived'].value_counts() plt.figure(figsize=(6, 4)) plt.bar(['Did Not Survive', 'Survived'], [counts[0], counts[1]], color=['red', 'green']) plt.title('Titanic — Survival Count') plt.xlabel('Status') plt.ylabel('Number of Passengers') plt.show() Always add title, xlabel, ylabel — never show a chart without labels!
Q21
Histogram — How are passenger ages distributed?
plt.figure(figsize=(8, 4)) plt.hist(df['Age'], bins=20, color='steelblue', edgecolor='white') plt.title('Age Distribution of Titanic Passengers') plt.xlabel('Age') plt.ylabel('Number of Passengers') plt.show() # Most passengers: 20–35 years old Histogram = how many people fall into each age range (bin)
Q22
Pie chart — Passenger class breakdown.
class_counts = df['Pclass'].value_counts() plt.figure(figsize=(5, 5)) plt.pie(class_counts, labels=['3rd Class', '1st Class', '2nd Class'], autopct='%1.1f%%', colors=['#FF6B6B', '#4ECDC4', '#45B7D1']) plt.title('Passengers by Class') plt.show()
Q23
Scatter plot — Does paying more = better survival?
survived = df[df['Survived'] == 1] not_survived = df[df['Survived'] == 0] plt.figure(figsize=(8, 5)) plt.scatter(not_survived['Age'], not_survived['Fare'], color='red', alpha=0.5, label='Did Not Survive') plt.scatter(survived['Age'], survived['Fare'], color='green', alpha=0.5, label='Survived') plt.title('Age vs Fare — Coloured by Survival') plt.xlabel('Age') plt.ylabel('Fare Paid') plt.legend() plt.show() alpha=0.5 makes dots semi-transparent so overlapping dots are visible
Q24
Bar chart — Survival rate by class (grouped bars).
survival_by_class = df.groupby('Pclass')['Survived'].mean() * 100 plt.figure(figsize=(6, 4)) plt.bar(['1st Class', '2nd Class', '3rd Class'], survival_by_class, color=['gold', 'silver', '#CD7F32']) plt.title('Survival Rate by Passenger Class (%)') plt.ylabel('Survival Rate (%)') plt.ylim(0, 100) plt.show() 1st: 63% 2nd: 47% 3rd: 24%
